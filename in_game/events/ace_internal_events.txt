# Auto Child Education - Internal events (hidden)
namespace = ace_internal

# Delayed education assignment for newborns (allows traits to be assigned first)
ace_internal.1 = {
	type = character_event
	hidden = yes
	
	trigger = {
		# Use is_child instead of has_trait_category (1.0.10 compatibility)
		is_child = yes
		owner = { 
			is_ai = no 
			NOT = { has_variable = ace_disabled }
		}
	}
	
	immediate = {
		if = {
			limit = { ace_should_auto_educate_child = yes }
			select_best_child_education_trait_based = yes
		}
	}
	
	option = { }
}
