# Auto Child Education - Hook into monthly country pulse with a delay
monthly_country_pulse = {
	on_actions = {
		delay = { days = 1 }
		auto_child_education_monthly_check
	}
}

# Our actual logic as a separate on_action
auto_child_education_monthly_check = {
	effect = {
		# Only run for human players
		if = {
			limit = { is_ai = no }
			# Check all children monthly
			every_character = {
				limit = {
					has_trait_category = child
					# Skip heirs entirely so player can manage heir education manually
					NOT = { is_heir = yes  }
					# Skip if already has expensive education
					NOT = { has_child_education = expensive_in_depth_education }
				}
				# Apply/update education based on current traits and aptitudes
				select_best_child_education_trait_based = yes
			}
		}
	}
}
